const newPass = document.getElementById("newPassword").value; if (newId && newPass) { let users = getUsers(); if (users.find(u => u.id === newId)) { alert("User ID already exists"); } else { users.push({ id: newId, pass: newPass }); saveUsers(users); alert("Account created successfully"); document.getElementById("newUserid").value = ""; document.getElementById("newPassword").value = ""; } } } function login() { const uid = document.getElementById("userid").value; const pwd = document.getElementById("password").value; const users = getUsers(); const found = users.find(u => u.id === uid && u.pass === pwd); if (found) { alert("Login successful"); document.getElementById("loginBox").style.display = "none"; document.getElementById("notifyForm").style.display = "block"; document.getElementById("notesForm").style.display = "block"; } else { alert("Invalid credentials"); } } function addNotification() { const text = document.getElementById("newNotification").value; if (text) { const div = document.createElement("div"); div.className = "notification-entry"; div.innerText = text; document.getElementById("notificationList").appendChild(div); document.getElementById("newNotification").value = ""; saveData(); } } function addNote() { const fileInput = document.getElementById("noteFile"); const file = fileInput.files[0]; if (file && file.type === "application/pdf") { const reader = new FileReader(); reader.onload = function(e) { const note = { name: file.name, data: e.target.result }; let notes = JSON.parse(localStorage.getItem("pdfNotes") || "[]"); notes.push(note); localStorage.setItem("pdfNotes", JSON.stringify(notes)); displayNote(note); fileInput.value = ""; }; reader.readAsDataURL(file); } else { alert("Please upload a valid PDF file."); } } function displayNote(note) { const div = document.createElement("div"); div.className = "note-entry"; const icon = document.createElement("img"); icon.src = "https://upload.wikimedia.org/wikipedia/commons/8/87/PDF_file_icon.svg"; icon.className = "pdf-icon"; const name = document.createElement("span"); name.textContent = note.name; const download = document.createElement("a"); download.href = note.data; download.download = note.name; download.textContent = "Download"; download.style.marginLeft = "10px"; div.appendChild(icon); div.appendChild(name); div.appendChild(download); document.getElementById("notesList").appendChild(div); } function saveData() { const notifications = Array.from(document.querySelectorAll(".notification-entry")).map(n => n.innerText); localStorage.setItem("notifications", JSON.stringify(notifications)); } function loadData() { const notifications = JSON.parse(localStorage.getItem("notifications") || "[]"); const notes = JSON.parse(localStorage.getItem("pdfNotes") || "[]"); notifications.forEach(text => { const div = document.createElement("div"); div.className = "notification-entry"; div.innerText = text; document.getElementById("notificationList").appendChild(div); }); notes.forEach(note => displayNote(note)); }
